87 static void fill_ranks(
88 const tatami::Matrix<double, int>* mat,
89 const std::vector<int>& universe,
90 int cell,
91 std::vector<double>& buffer,
92 tatami::Workspace* wrk,
93 RankedVector<double, int>& data_ranked)
94 {
95 data_ranked.clear();
96 if (universe.empty()) {
97 return;
98 }
99
100 size_t first = universe.front();
101 size_t last = universe.back() + 1;
102 auto ptr = mat->column(cell, buffer.data(), first, last, wrk);
103
104 for (auto u : universe) {
105 data_ranked.emplace_back(ptr[u - first], u);
106 }
107 std::sort(data_ranked.begin(), data_ranked.end());
108 return;

111 void prepare_mapping(
112 const IntegratedReference& ref,
113 const std::vector<int>& universe,
114 std::unordered_map<int, int>& mapping)
115 {
116 mapping.clear();
117 if (ref.check_availability) {
118 // If we need to check availability, we reconstruct the mapping
119 // for the intersection of features available in this reference.
120 // We then calculate the scaled ranks for the data.
121 int counter = 0;
122 for (auto c : universe) {
123 if (ref.available.find(c) != ref.available.end()) {
124 mapping[c] = counter;
125 ++counter;
126 }
127 }
128 } else {
129 // If we don't need to check availability, the mapping is
130 // much simpler. Technically, we could do this in the outer
131 // loop if none of the references required checks, but this
132 // seems like a niche optimization in practical settings.
133 for (size_t s = 0; s < universe.size(); ++s) {
134 auto u = universe[s];
135 mapping[u] = s;
136 }
137 }
138 return;
139 }

161 void run(
162 const tatami::Matrix<double, int>* mat,
163 const std::vector<const int*>& assigned,
164 const std::vector<IntegratedReference>& references,
165 int* best,
166 std::vector<double*>& scores,
167 double* delta)
168 {
169 size_t NR = mat->nrow();
170 size_t NC = mat->ncol();
171
172 #pragma omp parallel
173 {
174 std::vector<double> buffer(NR);
175 auto wrk = mat->new_workspace(false);
176
177 RankedVector<double, int> data_ranked, data_ranked2;
178 data_ranked.reserve(NR);
179 data_ranked2.reserve(NR);
180 RankedVector<int, int> ref_ranked;
181 ref_ranked.reserve(NR);
182
183 std::vector<double> scaled_data(NR);
184 std::vector<double> scaled_ref(NR);
185 std::unordered_set<int> universe_tmp;
186 std::vector<int> universe;
187 std::unordered_map<int, int> mapping;
188 std::vector<double> all_correlations;
189
190 #pragma omp for
191 for (size_t i = 0; i < NC; ++i) {
192 build_universe(i, assigned, references, universe_tmp, universe);
193 fill_ranks(mat, universe, i, buffer, wrk.get(), data_ranked);
194
195 // Scanning through each reference and computing the score for the best group.
196 double best_score = -1000, next_best = -1000;
197 int best_ref = 0;
198
199 for (size_t r = 0; r < references.size(); ++r) {
200 const auto& ref = references[r];
201 prepare_mapping(ref, universe, mapping);
202
203 scaled_ref.resize(mapping.size());
204 scaled_data.resize(mapping.size());
205
206 data_ranked2.clear();
207 subset_ranks(data_ranked, data_ranked2, mapping);
208 scaled_ranks(data_ranked2, scaled_data.data());
209
210 // Now actually calculating the score for the best group for
211 // this cell in this reference. This assumes that 'ref.ranked'
212 // already contains sorted pairs where the indices refer to the
213 // rows of the original data matrix.
214 auto best = assigned[r][i];
215 const auto& best_ranked = ref.ranked[best];
216 all_correlations.clear();
217
218 for (size_t s = 0; s < best_ranked.size(); ++s) {
219 ref_ranked.clear();
220 subset_ranks(best_ranked[s], ref_ranked, mapping);
221 scaled_ranks(ref_ranked, scaled_ref.data());
222 double cor = distance_to_correlation(scaled_ref.size(), scaled_data, scaled_ref);
223 all_correlations.push_back(cor);
224 }
225
226 double score = correlations_to_scores(all_correlations, quantile);
227 if (scores[r]) {
228 scores[r][i] = score;
229 }
230 if (score > best_score) {
231 next_best = best_score;
232 best_score = score;
233 best_ref = r;
234 } else if (score > next_best) {
235 next_best = score;
236 }
237 }
238
239 if (best) {
240 best[i] = best_ref;
241 }
242 if (delta && references.size() > 1) {
243 delta[i] = best_score - next_best;
244 }
245 }
246 }
247 return;
248 }
