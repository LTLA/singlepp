98 static void build_universe(int cell,
99 const std::vector<const int*>& assigned,

101 std::unordered_set<int>& uset,

104 uset.clear();

117 static void fill_ranks(

138 static void prepare_mapping(
139 const IntegratedReference& ref,
140 const std::vector<int>& universe,
141 std::unordered_map<int, int>& mapping)
142 {
143 mapping.clear();
144 if (ref.check_availability) {
145 // If we need to check availability, we reconstruct the mapping
146 // for the intersection of features available in this reference.
147 // We then calculate the scaled ranks for the data.
148 int counter = 0;
149 for (auto c : universe) {
150 if (ref.available.find(c) != ref.available.end()) {
151 mapping[c] = counter;
152 ++counter;
153 }
154 }
155 } else {
156 // If we don't need to check availability, the mapping is
157 // much simpler. Technically, we could do this in the outer
158 // loop if none of the references required checks, but this
159 // seems like a niche optimization in practical settings.
160 for (size_t s = 0; s < universe.size(); ++s) {
161 auto u = universe[s];
162 mapping[u] = s;
163 }
164 }
165 return;
166 }

188 void run(
189 const tatami::Matrix<double, int>* mat,
190 const std::vector<const int*>& assigned,
191 const std::vector<IntegratedReference>& references,
192 int* best,
193 std::vector<double*>& scores,
194 double* delta)
195 const {
196 auto NR = mat->nrow();
197
198 tatami::parallelize([&](int, int start, int len) -> void {
199 std::vector<double> buffer(NR);
200 auto wrk = tatami::consecutive_extractor<false, false>(mat, start, len);
201
202 RankedVector<double, int> data_ranked, data_ranked2;
203 data_ranked.reserve(NR);
204 data_ranked2.reserve(NR);
205 RankedVector<int, int> ref_ranked;
206 ref_ranked.reserve(NR);
207
208 std::vector<double> scaled_data(NR);
209 std::vector<double> scaled_ref(NR);
210 std::unordered_set<int> universe_tmp;
211 std::vector<int> universe;
212 std::unordered_map<int, int> mapping;
213 std::vector<double> all_correlations;
214
215 for (int i = start, end = start + len; i < end; ++i) {
216 build_universe(i, assigned, references, universe_tmp, universe);
217 fill_ranks(wrk.get(), universe, i, buffer, data_ranked);
218
219 // Scanning through each reference and computing the score for the best group.
220 double best_score = -1000, next_best = -1000;
221 int best_ref = 0;
222
223 for (size_t r = 0; r < references.size(); ++r) {
224 const auto& ref = references[r];
225 prepare_mapping(ref, universe, mapping);
226
227 scaled_ref.resize(mapping.size());
228 scaled_data.resize(mapping.size());
229
230 data_ranked2.clear();
231 subset_ranks(data_ranked, data_ranked2, mapping);
232 scaled_ranks(data_ranked2, scaled_data.data());
233
234 // Now actually calculating the score for the best group for
235 // this cell in this reference. This assumes that 'ref.ranked'
236 // already contains sorted pairs where the indices refer to the
237 // rows of the original data matrix.
238 auto best = assigned[r][i];
239 const auto& best_ranked = ref.ranked[best];
240 all_correlations.clear();
241
242 for (size_t s = 0; s < best_ranked.size(); ++s) {
243 ref_ranked.clear();
244 subset_ranks(best_ranked[s], ref_ranked, mapping);
245 scaled_ranks(ref_ranked, scaled_ref.data());
246 double cor = distance_to_correlation(scaled_ref.size(), scaled_data, scaled_ref);
247 all_correlations.push_back(cor);
248 }
249
250 double score = correlations_to_scores(all_correlations, quantile);
251 if (scores[r]) {
252 scores[r][i] = score;
253 }
254 if (score > best_score) {
255 next_best = best_score;
256 best_score = score;
257 best_ref = r;
258 } else if (score > next_best) {
259 next_best = score;
260 }
261 }
262
263 if (best) {
264 best[i] = best_ref;
265 }
266 if (delta && references.size() > 1) {
267 delta[i] = best_score - next_best;
268 }
269 }
270
271 }, mat->ncol(), nthreads);
272 }
