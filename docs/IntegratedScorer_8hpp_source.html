87 static void fill_ranks(
88 const tatami::Matrix<double, int>* mat,
89 const std::vector<int>& universe,
90 int cell,
91 std::vector<double>& buffer,
92 tatami::Workspace* wrk,

94 {
95 data_ranked.clear();
96 if (universe.empty()) {
97 return;
98 }
99
100 size_t first = universe.front();

102 auto ptr = mat->column(cell, buffer.data(), first, last, wrk);
103
104 for (auto u : universe) {
105 data_ranked.emplace_back(ptr[u - first], u);

111 void prepare_mapping(
112 const IntegratedReference& ref,
113 const std::vector<int>& universe,
114 std::unordered_map<int, int>& mapping)
115 {
116 mapping.clear();
117 if (ref.check_availability) {
118 // If we need to check availability, we reconstruct the mapping
119 // for the intersection of features available in this reference.
120 // We then calculate the scaled ranks for the data.
121 int counter = 0;
122 for (auto c : universe) {
123 if (ref.available.find(c) != ref.available.end()) {
124 mapping[c] = counter;
125 ++counter;
126 }
127 }
128 } else {
129 // If we don't need to check availability, the mapping is
130 // much simpler. Technically, we could do this in the outer
131 // loop if none of the references required checks, but this
132 // seems like a niche optimization in practical settings.
133 for (size_t s = 0; s < universe.size(); ++s) {
134 auto u = universe[s];
135 mapping[u] = s;
136 }
137 }
138 return;
139 }
