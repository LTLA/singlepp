158 static void fill_ranks(
159 const tatami::Matrix<double, int>* mat,
160 const std::vector<int>& universe,
161 int cell,
162 std::vector<double>& buffer,
163 tatami::Workspace* wrk,
164 RankedVector<double, int>& data_ranked)
165 {
166 data_ranked.clear();
167 if (universe.empty()) {
168 return;
169 }
170
171 size_t first = universe.front();
172 size_t last = universe.back() + 1;
173 auto ptr = mat->column(cell, buffer.data(), first, last, wrk);
174
175 for (auto u : universe) {
176 data_ranked.emplace_back(ptr[u - first], u);
177 }
178 std::sort(data_ranked.begin(), data_ranked.end());
179 return;

182 void prepare_mapping(
183 const IntegratedReference& ref,
184 const std::vector<int>& universe,
185 std::unordered_map<int, int>& mapping)
186 {
187 mapping.clear();
188 if (ref.check_availability) {
189 // If we need to check availability, we reconstruct the mapping
190 // for the intersection of features available in this reference.
191 // We then calculate the scaled ranks for the data.
192 int counter = 0;
193 for (auto c : universe) {
194 if (ref.available.find(c) != ref.available.end()) {
195 mapping[c] = counter;
196 ++counter;
197 }
198 }
199 } else {
200 // If we don't need to check availability, the mapping is
201 // much simpler. Technically, we could do this in the outer
202 // loop if none of the references required checks, but this
203 // seems like a niche optimization in practical settings.
204 for (size_t s = 0; s < universe.size(); ++s) {
205 auto u = universe[s];
206 mapping[u] = s;
207 }
208 }
209 return;
210 }
