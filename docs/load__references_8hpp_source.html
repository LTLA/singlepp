36template<bool parallel_, class Reader_>
37std::vector<int> load_labels_internal(Reader_& reader) {
38 bool non_empty = false;
39 int current = 0;
40 std::vector<int> labels;
41 bool remaining = true;
42
43 SuperPerByte<parallel_> pb(&reader);
44 bool okay = pb.valid();
45 while (okay) {
46 char x = pb.get();
47 okay = pb.advance();
48
49 if (x == '\n') {
50 if (!non_empty) {
51 throw std::runtime_error("label index must be an integer");
52 }
53 labels.push_back(current);
54 current = 0;
55 non_empty = false;
56 } else if (std::isdigit(x)) {
57 non_empty = true;
58 current *= 10;
59 current += (x - '0');
60 } else {
61 throw std::runtime_error("label index must be an integer");
62 }
63 }
64
65 if (non_empty) {
66 labels.push_back(current);
67 }
68
69 return labels;
70}

134template<bool parallel_, class Reader_>
135std::vector<std::string> load_names_internal(Reader_& reader) {
136 std::string current;
137 std::vector<std::string> names;
138
139 SuperPerByte<parallel_> pb(&reader);
140 bool okay = pb.valid();
141 while (okay) {
142 char x = pb.get();
143 okay = pb.advance();
144
145 if (x == '\n') {
146 names.emplace_back(std::move(current));
147 current.clear();
148 } else {
149 current += x;
150 }
151 }
152
153 if (!current.empty()) { // absence of trailing newline is okay.
154 names.emplace_back(std::move(current));
155 }
156
157 return names;

221template<bool parallel_, class Reader_>
222std::pair<std::vector<std::string>, std::vector<std::string> > load_features_internal(Reader_& reader) {
223 std::string current;
224 std::vector<std::string> ensembl, symbols;
225
226 SuperPerByte<parallel_> pb(&reader);
227 bool okay = pb.valid();
228 while (okay) {
229 current.clear();
230
231 // Pulling out the Ensembl ID.
232 do {
233 char x = pb.get();
234 if (x == ',') { // don't advance yet, so that okay check below doesn't trigger if the symbol is empty and file is not newline terminated.
235 break;
236 } else if (x == '\n') {
237 okay = false; // hit the error below.
238 break;
239 }
240 current += x;
241 okay = pb.advance();
242 } while (okay);
243
244 if (!okay) {
245 throw std::runtime_error("two comma-separated fields (Ensembl ID and symbol) expected on each line");
246 }
247 okay = pb.advance();
248
249 ensembl.push_back(std::move(current));
250 current.clear();
251
252 // Now pulling out the gene symbol.
253 while (okay) {
254 char x = pb.get();
255 okay = pb.advance();
256 if (x == '\n') {
257 break;
258 }
259 current += x;
260 }
261
262 symbols.push_back(std::move(current)); // still gets added if file is not newline terminated.
263 current.clear();
264 }
265
266 return std::make_pair(std::move(ensembl), std::move(symbols));
267}

345template<typename Data, typename Index, bool parallel_, class Reader>
346RankMatrix<Data, Index> load_rankings_internal(Reader& reader) {
347 size_t nfeatures = 0;
348 size_t line = 0;
349 std::vector<int> values;
350
351 int field = 0;
352 bool non_empty = false;
353 int current = 0;
354
355 bool has_nfeatures = false;
356 auto check_nfeatures = [&]() -> void {
357 if (!has_nfeatures) {
358 has_nfeatures = true;
359 nfeatures = field + 1;
360 } else if (field + 1 != nfeatures) {
361 throw std::runtime_error("number of fields on each line should be equal to the number of features");
362 }
363 };
364
365 SuperPerByte<parallel_> pb(&reader);
366 bool okay = pb.valid();
367 while (okay) {
368 char x = pb.get();
369 okay = pb.advance();
370
371 if (x == '\n') {
372 check_nfeatures();
373 if (!non_empty) {
374 throw std::runtime_error("fields should not be empty");
375 }
376 values.push_back(current);
377 current = 0;
378 field = 0;
379 non_empty = false;
380 ++line;
381
382 } else if (x == ',') {
383 if (!non_empty) {
384 throw std::runtime_error("fields should not be empty");
385 }
386 values.push_back(current);
387 current = 0;
388 ++field;
389 non_empty = false;
390
391 } else if (std::isdigit(x)) {
392 non_empty = true;
393 current *= 10;
394 current += (x - '0');
395
396 } else {
397 throw std::runtime_error("fields should only contain integer ranks");
398 }
399 }
400
401 if (field || non_empty) { // aka no terminating newline.
402 check_nfeatures();
403 if (!non_empty) {
404 throw std::runtime_error("fields should not be empty");
405 }
406 values.push_back(current);
407 ++line;
408 }
409
410 return RankMatrix<Data, Index>(nfeatures, line, std::move(values));
411};

484template<bool parallel_, class Reader>
485Markers load_markers_internal(size_t nfeatures, size_t nlabels, Reader& reader) {
486 Markers markers(nlabels);
487 for (auto& m : markers) {
488 m.resize(nlabels);
489 }
490
491 std::vector<int> values;
492 SuperPerByte<parallel_> pb(&reader);
493 bool okay = pb.valid();
494 while (okay) {
495
496 // Processing the label IDs.
497 size_t first = 0, second = 0;
498 for (int l = 0; l < 2; ++l) {
499 auto& current = (l == 0 ? first : second);
500 bool non_empty = false;
501
502 do {
503 char x = pb.get();
504 okay = pb.advance();
505
506 if (x == '\t') {
507 if (!non_empty) {
508 throw std::runtime_error("empty field detected in the label indices");
509 }
510 break;
511 } else if (x == '\n') {
512 okay = false; // hit the error below.
513 break;
514 } else if (!std::isdigit(x)) {
515 throw std::runtime_error("label indices should be integers");
516 }
517
518 non_empty = true;
519 current *= 10;
520 current += (x - '0');
521 } while (okay);
522
523 if (!okay) {
524 throw std::runtime_error("expected at least three tab-separated fields on each line");
525 }
526 if (current >= markers.size()) {
527 throw std::runtime_error("label index out of range");
528 }
529 }
530
531 // Processing the actual gene indices.
532 bool non_empty = false;
533 int current = 0;
534 while (okay) {
535 char x = pb.get();
536 okay = pb.advance();
537
538 if (std::isdigit(x)) {
539 non_empty = true;
540 current *= 10;
541 current += (x - '0');
542
543 } else if (x == '\t') {
544 if (!non_empty) {
545 throw std::runtime_error("gene index fields should not be empty");
546 }
547 values.push_back(current);
548 current = 0;
549 non_empty = false;
550
551 } else if (x == '\n') {
552 break;
553
554 } else {
555 throw std::runtime_error("gene index fields should be integers");
556 }
557 }
558
559 // Adding the last element. We don't do this inside the newline check,
560 // as we need to account for cases where the file is not newline-terminated.
561 if (!non_empty) {
562 throw std::runtime_error("gene index fields should not be empty");
563 }
564 values.push_back(current);
565
566 for (auto v : values) {
567 if (static_cast<size_t>(v) >= nfeatures) {
568 throw std::runtime_error("gene index out of range");
569 }
570 }
571
572 auto& store = markers[first][second];
573 if (!store.empty()) {
574 throw std::runtime_error("multiple marker sets listed for a single pairwise comparison");
575 }
576 store.swap(values); // implicit clear of 'values', as store is empty.
577 }
578
579 return markers;
580}
