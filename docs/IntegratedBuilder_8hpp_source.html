48 template<class Subset>
49 void add_internal(const tatami::Matrix<double, int>* ref, const int* labels, const Markers& old_markers, const Subset& subset) {
50 add_internal(ref, labels);
51
52 // Adding the markers for each label, indexed according to their
53 // position in the test matrix. This assumes that 'subset' is
54 // appropriately specified to contain the test's row indices. Note that
55 // these are positions in the test, not the ref; IntegratedScorer will
56 // use the gene_mapping to convert them to row indices of 'ref'.
57 auto& new_markers = references.back().markers;
58 new_markers.reserve(old_markers.size());
59
60 for (size_t i = 0; i < old_markers.size(); ++i) {
61 const auto& cur_old_markers = old_markers[i];
62
63 std::unordered_set<int> unified;
64 for (const auto& x : cur_old_markers) {
65 unified.insert(x.begin(), x.end());
66 }
67
68 new_markers.emplace_back(unified.begin(), unified.end());
69
70 if constexpr(!std::is_same<Subset, bool>::value) {
71 auto& cur_new_markers = new_markers.back();
72 for (auto& y : cur_new_markers) {
73 y = subset[y];
74 }
75 }
76 }
77 return;
78 }

80 template<typename Id, class Subset>
81 void add_internal(size_t mat_nrow,
82 const Id* mat_id,
83 const tatami::Matrix<double, int>* ref,
84 const Id* ref_id,
85 const int* labels,
86 const Markers& old_markers,
87 const Subset& subset)
88 {
89 add_internal(ref, labels);
90
91 auto intersection = intersect_features(mat_nrow, mat_id, ref->nrow(), ref_id);
92
93 // Manually constructing the markers. This involves (i) pruning out the
94 // markers that aren't present in the intersection, and (ii) updating
95 // their indices so that they point to rows of 'mat', not 'ref'.
96 std::unordered_map<int, int> reverse_map;
97 for (const auto& i : intersection) {
98 reverse_map[i.second] = i.first;
99 }
100
101 auto subindex = [&](int i) -> int {
102 if constexpr(!std::is_same<Subset, bool>::value) {
103 return subset[i];
104 } else {
105 return i;
106 }
107 };
108
109 auto& new_markers = references.back().markers;
110 new_markers.resize(old_markers.size());
111
112 for (size_t i = 0; i < old_markers.size(); ++i) {
113 const auto& cur_old_markers = old_markers[i];
114
115 std::unordered_set<int> unified;
116 for (const auto& x : cur_old_markers) {
117 unified.insert(x.begin(), x.end());
118 }
119
120 auto& cur_new_markers = new_markers[i];
121 cur_new_markers.reserve(unified.size());
122
123 for (auto y : unified) {
124 auto it = reverse_map.find(subindex(y));
125 if (it != reverse_map.end()) {
126 cur_new_markers.push_back(it->second);
127 }
128 }
129 }
130
131 // Constructing the mapping of mat's rows to the reference rows.
132 references.back().check_availability = true;
133 auto& mapping = gene_mapping.back();
134 for (const auto& i : intersection) {
135 mapping[i.first] = i.second;
136 }
137 return;
138 }

301 size_t NR = curmat->nrow();

309 positions[c] = pos;
310 ++pos;
311 }

313 curref.ranked.resize(nlabels);

315 curref.ranked[l].resize(samples_per_label[l]);
316 }
317

320 // the reference has 1:1 feature mapping to the test dataset.
321 // In that case, we can proceed quite simply.
