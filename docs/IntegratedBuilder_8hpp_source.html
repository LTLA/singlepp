62 void add(const tatami::Matrix<double, int>* ref, const int* labels, const Markers& old_markers, const std::vector<int>& subset) {
63 add(ref, labels);
64
65 // Adding the markers for each label, indexed according to their
66 // position in the test matrix. This assumes that 'subset' is
67 // appropriately specified to contain the test's row indices. Note that
68 // these are positions in the test, not the ref; IntegratedScorer will
69 // use the gene_mapping to convert them to row indices of 'ref'.
70 auto& new_markers = references.back().markers;
71 new_markers.reserve(old_markers.size());
72
73 for (size_t i = 0; i < old_markers.size(); ++i) {
74 const auto& cur_old_markers = old_markers[i];
75
76 std::unordered_set<int> unified;
77 for (const auto& x : cur_old_markers) {
78 unified.insert(x.begin(), x.end());
79 }
80
81 new_markers.emplace_back(unified.begin(), unified.end());
82 auto& cur_new_markers = new_markers.back();
83 for (auto& y : cur_new_markers) {
84 y = subset[y];
85 }
86 }
87 return;

220 std::vector<IntegratedReference> finish() {
221 // Identify the global set of all genes that will be in use here.
222 std::unordered_set<int> in_use_tmp;
223 for (const auto& ref : references) {
224 for (const auto& mrk : ref.markers) {
225 in_use_tmp.insert(mrk.begin(), mrk.end());
226 }
227 }
228
229 std::vector<int> in_use(in_use_tmp.begin(), in_use_tmp.end());
230 std::sort(in_use.begin(), in_use.end());
231
232 for (size_t r = 0; r < references.size(); ++r) {
233 auto& curref = references[r];
234 auto curlab = stored_labels[r];
235 auto curmat = stored_matrices[r];
236
237 size_t NR = curmat->nrow();
238 size_t NC = curmat->ncol();
239 size_t nlabels = curref.markers.size();
240
241 std::vector<int> positions(NC);
242 std::vector<int> samples_per_label(nlabels);
243 for (size_t c = 0; c < NC; ++c) {
244 auto& pos = samples_per_label[curlab[c]];
245 positions[c] = pos;
246 ++pos;
247 }
248
249 curref.ranked.resize(nlabels);
250 for (size_t l = 0; l < nlabels; ++l) {
251 curref.ranked[l].resize(samples_per_label[l]);
252 }
253
254 if (!curref.check_availability) {
255 // If we don't need to check availability, this implies that
256 // the reference has 1:1 feature mapping to the test dataset.
257 // In that case, we can proceed quite simply.
258 size_t first = 0, last = 0;
259 if (in_use.size()) {
260 first = in_use.front();
261 last = in_use.back() + 1;
262 }
263
264 #pragma omp parallel
265 {
266 RankedVector<double, int> tmp_ranked;
267 tmp_ranked.reserve(in_use.size());
268 std::vector<double> buffer(NR);
269 auto wrk = curmat->new_workspace(false);
270
271 #pragma omp for
272 for (size_t c = 0; c < NC; ++c) {
273 auto ptr = curmat->column(c, buffer.data(), first, last, wrk.get());
274
275 tmp_ranked.clear();
276 for (auto u : in_use) {
277 tmp_ranked.emplace_back(ptr[u - first], u);
278 }
279 std::sort(tmp_ranked.begin(), tmp_ranked.end());
280
281 auto& final_ranked = curref.ranked[curlab[c]][positions[c]];
282 simplify_ranks(tmp_ranked, final_ranked);
283 }
284 }
285
286 } else {
287 // If we do need to check availability, then we need to check
288 // the mapping of test genes to their reference row indices.
289 const auto& cur_mapping = gene_mapping[r];
290 auto& cur_available = curref.available;
291 std::unordered_map<int, int> remapping;
292 remapping.reserve(in_use.size());
293 size_t first = NR, last = 0;
294
295 for (auto u : in_use) {
296 auto it = cur_mapping.find(u);
297 if (it == cur_mapping.end()) {
298 continue;
299 }
300
301 remapping[u] = it->second;
302 cur_available.insert(u);
303 if (it->second < first) {
304 first = it->second;
305 }
306 if (it->second > last) {
307 last = it->second;
308 }
309 }
310 last = std::max(last + 1, first);
311
312 #pragma omp parallel
313 {
314 RankedVector<double, int> tmp_ranked;
315 tmp_ranked.reserve(in_use.size());
316 std::vector<double> buffer(NR);
317 auto wrk = curmat->new_workspace(false);
318
319 #pragma omp for
320 for (size_t c = 0; c < NC; ++c) {
321 auto ptr = curmat->column(c, buffer.data(), first, last, wrk.get());
322
323 tmp_ranked.clear();
324 for (auto p : remapping) {
325 tmp_ranked.emplace_back(ptr[p.second - first], p.first);
326 }
327 std::sort(tmp_ranked.begin(), tmp_ranked.end());
328
329 auto& final_ranked = curref.ranked[curlab[c]][positions[c]];
330 simplify_ranks(tmp_ranked, final_ranked);
331 }
332 }
333 }
334 }
335
336 return std::move(references);
